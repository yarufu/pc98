#pal adv.pal
#bg class
#bgm PLAY.M
#right C01_N.SPR

[サンプル子]

#label classroom

#ifeq progress 0 classroom0
#ifeq progress 1 classroom1
#ifeq progress 2 classroom2


;
; #label classroom シーン0
;

#label classroom0

#choice

見る
話す
移動

#endchoice

#ifeq choice 1 classroom0_look
#ifeq choice 2 classroom0_talk
#ifeq choice 3 classroom0_move

;見る
#label classroom0_look
#call common_look
#jump classroom

;話す
#label classroom0_talk
